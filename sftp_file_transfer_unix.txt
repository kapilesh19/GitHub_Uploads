#!/bin/bash
# =============================================================
#  SFTP File Transfer Script
#  Usage : ./sftp_transfer.sh --folder /path/to/folder
#
#  - One src server, one tgt server
#  - Only 2 password prompts (once for src, once for tgt)
#  - Two cmds files generated dynamically:
#      cmds_src.txt  : check + get all files from src
#      cmds_tgt.txt  : check + backup + put all files to tgt
#
#  Folder must contain:
#    files.csv : src_server,src_path,src_filename,tgt_server,tgt_path
#    creds.csv : server_name,port,username
#
#  Output:
#    summary_<YYYYMMDD_HHMMSS>.csv
#    sftp_transfer_<YYYYMMDD_HHMMSS>.log
# =============================================================

TIMESTAMP=$(date '+%Y%m%d_%H%M%S')
TODAY=$(date '+%Y%m%d')

# ─────────────────────────────────────────────
#  Parse --folder argument
# ─────────────────────────────────────────────
FOLDER=""
while [[ $# -gt 0 ]]; do
    case "$1" in
        --folder|-f)
            FOLDER="$2"
            shift 2
            ;;
        *)
            echo "Unknown argument: $1"
            echo "Usage: $0 --folder /path/to/folder"
            exit 1
            ;;
    esac
done

if [[ -z "$FOLDER" ]]; then
    echo "ERROR: --folder argument is required"
    echo "Usage: $0 --folder /path/to/folder"
    exit 1
fi

FOLDER=$(cd "$FOLDER" && pwd)

if [[ ! -d "$FOLDER" ]]; then
    echo "ERROR: Folder not found: $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  File paths
# ─────────────────────────────────────────────
FILES_CSV="$FOLDER/files.csv"
CREDS_CSV="$FOLDER/creds.csv"
SUMMARY_CSV="$FOLDER/summary_${TIMESTAMP}.csv"
LOG_FILE="$FOLDER/sftp_transfer_${TIMESTAMP}.log"
TMP_DIR="$FOLDER/tmp_${TIMESTAMP}"
CMDS_SRC="$TMP_DIR/cmds_src.txt"
CMDS_TGT="$TMP_DIR/cmds_tgt.txt"

mkdir -p "$TMP_DIR"

# ─────────────────────────────────────────────
#  Validate input files
# ─────────────────────────────────────────────
if [[ ! -f "$FILES_CSV" ]]; then
    echo "ERROR: files.csv not found in $FOLDER"
    exit 1
fi
if [[ ! -f "$CREDS_CSV" ]]; then
    echo "ERROR: creds.csv not found in $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  Log function
# ─────────────────────────────────────────────
log() {
    local msg="$(date '+%Y-%m-%d %H:%M:%S') [$1] $2"
    echo "$msg"
    echo "$msg" >> "$LOG_FILE"
}

# ─────────────────────────────────────────────
#  Get a field from creds.csv for a server
#  Usage : get_cred <server> <field>
#  Fields: port=2  username=3
# ─────────────────────────────────────────────
get_cred() {
    local search_server=$(echo "$1" | tr '[:upper:]' '[:lower:]')
    local field="$2"
    local col
    case "$field" in
        port)     col=2 ;;
        username) col=3 ;;
    esac

    local first=1
    while IFS=',' read -r srv p u; do
        if [[ $first -eq 1 ]]; then first=0; continue; fi
        srv=$(echo "$srv" | tr -d ' ' | tr '[:upper:]' '[:lower:]')
        if [[ "$srv" == "$search_server" ]]; then
            case "$col" in
                2) echo $(echo "$p" | tr -d ' ') ;;
                3) echo $(echo "$u" | tr -d ' ') ;;
            esac
            return
        fi
    done < "$CREDS_CSV"
}

# ─────────────────────────────────────────────
#  Build backup filename
#  report.csv -> report_20240315.csv
# ─────────────────────────────────────────────
build_backup_name() {
    local filename="$1"
    if [[ "$filename" == *.* ]]; then
        echo "${filename%.*}_${TODAY}.${filename##*.}"
    else
        echo "${filename}_${TODAY}"
    fi
}

# ─────────────────────────────────────────────
#  Convert bytes to KB
# ─────────────────────────────────────────────
bytes_to_kb() {
    local bytes="$1"
    if [[ -z "$bytes" ]]; then echo "N/A"; return; fi
    echo "scale=2; $bytes / 1024" | bc
}

# ─────────────────────────────────────────────
#  SINGLE SFTP FUNCTION
#  Usage : run_sftp <port> <user> <server> <cmds_file>
#  Returns 0 on success, 1 on failure
#  sftp will naturally prompt for password
# ─────────────────────────────────────────────
run_sftp() {
    local port="$1"
    local user="$2"
    local server="$3"
    local cmds_file="$4"

    sftp -P "$port" "${user}@${server}" < "$cmds_file"
    return $?
}

# ─────────────────────────────────────────────
#  Read files.csv and collect all rows
#  Also extract src/tgt server and credentials
# ─────────────────────────────────────────────
declare -a ALL_SRC_FILENAMES
declare -a ALL_SRC_PATHS
declare -a ALL_TGT_PATHS

SRC_SERVER=""
TGT_SERVER=""

first_line=1
while IFS=',' read -r src_server src_path src_filename tgt_server tgt_path; do
    if [[ $first_line -eq 1 ]]; then first_line=0; continue; fi

    src_server=$(echo "$src_server"     | tr -d ' ')
    src_path=$(echo "$src_path"         | tr -d ' ')
    src_filename=$(echo "$src_filename" | tr -d ' ')
    tgt_server=$(echo "$tgt_server"     | tr -d ' ')
    tgt_path=$(echo "$tgt_path"         | tr -d ' ')

    src_path="${src_path%/}"
    tgt_path="${tgt_path%/}"

    # Capture server names from first data row
    if [[ -z "$SRC_SERVER" ]]; then SRC_SERVER="$src_server"; fi
    if [[ -z "$TGT_SERVER" ]]; then TGT_SERVER="$tgt_server"; fi

    ALL_SRC_FILENAMES+=("$src_filename")
    ALL_SRC_PATHS+=("$src_path")
    ALL_TGT_PATHS+=("$tgt_path")

done < "$FILES_CSV"

TOTAL=${#ALL_SRC_FILENAMES[@]}

log "INFO" "============================================================"
log "INFO" "SFTP File Transfer Script"
log "INFO" "Run date     : $TODAY"
log "INFO" "Folder       : $FOLDER"
log "INFO" "Total files  : $TOTAL"
log "INFO" "SRC server   : $SRC_SERVER"
log "INFO" "TGT server   : $TGT_SERVER"
log "INFO" "============================================================"

# ─────────────────────────────────────────────
#  Get credentials
# ─────────────────────────────────────────────
SRC_PORT=$(get_cred "$SRC_SERVER" "port");  SRC_PORT=${SRC_PORT:-22}
SRC_USER=$(get_cred "$SRC_SERVER" "username")
TGT_PORT=$(get_cred "$TGT_SERVER" "port");  TGT_PORT=${TGT_PORT:-22}
TGT_USER=$(get_cred "$TGT_SERVER" "username")

if [[ -z "$SRC_USER" ]]; then
    log "ERROR" "No credentials found for SRC server: $SRC_SERVER"
    exit 1
fi
if [[ -z "$TGT_USER" ]]; then
    log "ERROR" "No credentials found for TGT server: $TGT_SERVER"
    exit 1
fi

# ─────────────────────────────────────────────
#  Write summary CSV header
# ─────────────────────────────────────────────
echo "filename,available_in_src,available_in_tgt,backup_filename_in_tgt,file_copied_src_to_tgt,src_file_size_kb,tgt_file_size_kb,backed_up_file_size_kb" \
    > "$SUMMARY_CSV"

# =============================================================
#  STEP 1 — BUILD cmds_src.txt
#  For each file:
#    ls -l  : check file exists and has read permission
#    get    : pull file to local TMP_DIR
# =============================================================
log "INFO" "Building $CMDS_SRC ..."

> "$CMDS_SRC"   # create empty file

for i in "${!ALL_SRC_FILENAMES[@]}"; do
    src_filename="${ALL_SRC_FILENAMES[$i]}"
    src_path="${ALL_SRC_PATHS[$i]}"
    src_full="${src_path}/${src_filename}"
    tmp_local="${TMP_DIR}/${src_filename}"

    echo "ls -l ${src_full}"          >> "$CMDS_SRC"
    echo "get ${src_full} ${tmp_local}" >> "$CMDS_SRC"
done

echo "bye" >> "$CMDS_SRC"

log "INFO" "Generated $CMDS_SRC with $TOTAL file(s)"
log "INFO" "Contents of $CMDS_SRC :"
cat "$CMDS_SRC" >> "$LOG_FILE"

# =============================================================
#  STEP 2 — RUN cmds_src.txt  (PASSWORD PROMPT 1 — SRC)
#  sftp will prompt: user@srcserver's password:
# =============================================================
log "INFO" "------------------------------------------------------------"
log "INFO" "Connecting to SRC: ${SRC_USER}@${SRC_SERVER}:${SRC_PORT}"
log "INFO" "------------------------------------------------------------"

run_sftp "$SRC_PORT" "$SRC_USER" "$SRC_SERVER" "$CMDS_SRC"
SRC_RC=$?

if [[ $SRC_RC -ne 0 ]]; then
    log "ERROR" "SFTP to SRC failed (exit code: $SRC_RC). Aborting."
    rm -rf "$TMP_DIR"
    exit 1
fi

# ─────────────────────────────────────────────
#  Verify all files were actually downloaded
#  and have read permission
#  Abort if any file is missing or unreadable
# ─────────────────────────────────────────────
log "INFO" "Verifying downloaded files..."

for i in "${!ALL_SRC_FILENAMES[@]}"; do
    src_filename="${ALL_SRC_FILENAMES[$i]}"
    tmp_local="${TMP_DIR}/${src_filename}"

    if [[ ! -f "$tmp_local" ]]; then
        log "ERROR" "File not downloaded: $src_filename — aborting."
        rm -rf "$TMP_DIR"
        exit 1
    fi

    if [[ ! -r "$tmp_local" ]]; then
        log "ERROR" "File not readable: $src_filename — aborting."
        rm -rf "$TMP_DIR"
        exit 1
    fi

    src_size=$(wc -c < "$tmp_local" | tr -d ' ')
    log "INFO" "  OK: $src_filename ($(bytes_to_kb $src_size) KB)"
done

log "INFO" "All $TOTAL file(s) downloaded successfully from SRC."

# =============================================================
#  STEP 3 — BUILD cmds_tgt.txt
#  For each file:
#    ls -l           : check if file exists on tgt
#    rename          : backup existing file with timestamp
#    put             : upload new file from local to tgt
# =============================================================
log "INFO" "Building $CMDS_TGT ..."

> "$CMDS_TGT"   # create empty file

for i in "${!ALL_SRC_FILENAMES[@]}"; do
    src_filename="${ALL_SRC_FILENAMES[$i]}"
    tgt_path="${ALL_TGT_PATHS[$i]}"
    tgt_full="${tgt_path}/${src_filename}"
    bak_name=$(build_backup_name "$src_filename")
    bak_full="${tgt_path}/${bak_name}"
    tmp_local="${TMP_DIR}/${src_filename}"

    echo "ls -l ${tgt_full}"               >> "$CMDS_TGT"
    echo "-rm ${bak_full}"                 >> "$CMDS_TGT"   # remove old backup if exists (- ignores error)
    echo "rename ${tgt_full} ${bak_full}"  >> "$CMDS_TGT"   # backup current file (ignored if not exists)
    echo "put ${tmp_local} ${tgt_full}"    >> "$CMDS_TGT"   # upload new file
done

echo "bye" >> "$CMDS_TGT"

log "INFO" "Generated $CMDS_TGT with $TOTAL file(s)"
log "INFO" "Contents of $CMDS_TGT :"
cat "$CMDS_TGT" >> "$LOG_FILE"

# =============================================================
#  STEP 4 — RUN cmds_tgt.txt  (PASSWORD PROMPT 2 — TGT)
#  sftp will prompt: user@tgtserver's password:
# =============================================================
log "INFO" "------------------------------------------------------------"
log "INFO" "Connecting to TGT: ${TGT_USER}@${TGT_SERVER}:${TGT_PORT}"
log "INFO" "------------------------------------------------------------"

run_sftp "$TGT_PORT" "$TGT_USER" "$TGT_SERVER" "$CMDS_TGT"
TGT_RC=$?

if [[ $TGT_RC -ne 0 ]]; then
    log "ERROR" "SFTP to TGT failed (exit code: $TGT_RC). Aborting."
    rm -rf "$TMP_DIR"
    exit 1
fi

log "INFO" "All files transferred to TGT successfully."

# =============================================================
#  STEP 5 — BUILD SUMMARY CSV
#  Gather sizes from local temp files (already downloaded)
# =============================================================
log "INFO" "Building summary..."

for i in "${!ALL_SRC_FILENAMES[@]}"; do
    src_filename="${ALL_SRC_FILENAMES[$i]}"
    tgt_path="${ALL_TGT_PATHS[$i]}"
    bak_name=$(build_backup_name "$src_filename")
    tmp_local="${TMP_DIR}/${src_filename}"

    # Get src file size from local temp copy
    if [[ -f "$tmp_local" ]]; then
        src_bytes=$(wc -c < "$tmp_local" | tr -d ' ')
        src_size_kb=$(bytes_to_kb "$src_bytes")
        avail_src="Yes"
    else
        src_size_kb="N/A"
        avail_src="No"
    fi

    # We attempted transfer so mark these accordingly
    avail_tgt="Yes"
    backup_name="$bak_name"
    copied_flag="Yes"
    tgt_size_kb="$src_size_kb"   # same size as src
    bak_size_kb="$src_size_kb"   # backup was the previous file (same size assumed)

    echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" \
        >> "$SUMMARY_CSV"
done

# ─────────────────────────────────────────────
#  Cleanup temp folder
# ─────────────────────────────────────────────
rm -rf "$TMP_DIR"

# ─────────────────────────────────────────────
#  Final results
# ─────────────────────────────────────────────
log "INFO" "============================================================"
log "INFO" "RESULTS  : Total=$TOTAL | All files copied successfully"
log "INFO" "Summary  : $SUMMARY_CSV"
log "INFO" "Log      : $LOG_FILE"
log "INFO" "============================================================"
