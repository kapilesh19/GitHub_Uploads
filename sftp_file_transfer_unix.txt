#!/bin/bash
# =============================================================
#  SFTP File Transfer Script
#  Usage : ./sftp_transfer.sh --folder /path/to/folder
#
#  Folder must contain:
#    files.csv : src_server,src_path,src_filename,tgt_server,tgt_path
#    creds.csv : server_name,port,username,password
#
#  Output files generated in same folder:
#    cmds_src_<filename>.txt
#    cmds_tgt_<filename>.txt
#    cmds_tgt_rename_<filename>.txt
#    summary_<YYYYMMDD_HHMMSS>.csv
#    sftp_transfer_<YYYYMMDD_HHMMSS>.log
# =============================================================

TIMESTAMP=$(date '+%Y%m%d_%H%M%S')
TODAY=$(date '+%Y%m%d')

# ─────────────────────────────────────────────
#  Parse --folder argument
# ─────────────────────────────────────────────
FOLDER=""
while [[ $# -gt 0 ]]; do
    case "$1" in
        --folder|-f)
            FOLDER="$2"
            shift 2
            ;;
        *)
            echo "Unknown argument: $1"
            echo "Usage: $0 --folder /path/to/folder"
            exit 1
            ;;
    esac
done

if [[ -z "$FOLDER" ]]; then
    echo "ERROR: --folder argument is required"
    echo "Usage: $0 --folder /path/to/folder"
    exit 1
fi

# Convert to absolute path
FOLDER=$(cd "$FOLDER" && pwd)

if [[ ! -d "$FOLDER" ]]; then
    echo "ERROR: Folder not found: $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  File paths
# ─────────────────────────────────────────────
FILES_CSV="$FOLDER/files.csv"
CREDS_CSV="$FOLDER/creds.csv"
SUMMARY_CSV="$FOLDER/summary_${TIMESTAMP}.csv"
LOG_FILE="$FOLDER/sftp_transfer_${TIMESTAMP}.log"
TMP_DIR="$FOLDER/tmp_${TIMESTAMP}"

mkdir -p "$TMP_DIR"

# ─────────────────────────────────────────────
#  Validate input files
# ─────────────────────────────────────────────
if [[ ! -f "$FILES_CSV" ]]; then
    echo "ERROR: files.csv not found in $FOLDER"
    exit 1
fi

if [[ ! -f "$CREDS_CSV" ]]; then
    echo "ERROR: creds.csv not found in $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  Log function
# ─────────────────────────────────────────────
log() {
    local msg="$(date '+%Y-%m-%d %H:%M:%S') [$1] $2"
    echo "$msg"
    echo "$msg" >> "$LOG_FILE"
}

# ─────────────────────────────────────────────
#  Get a field from creds.csv for a server
#  Usage: get_cred <server> <field>
#  Field: port=2, username=3, password=4
# ─────────────────────────────────────────────
get_cred() {
    local search_server=$(echo "$1" | tr '[:upper:]' '[:lower:]')
    local field="$2"
    local col

    case "$field" in
        port)     col=2 ;;
        username) col=3 ;;
        password) col=4 ;;
    esac

    # Read creds.csv line by line skipping header
    local first=1
    while IFS=',' read -r srv p u pw; do
        if [[ $first -eq 1 ]]; then
            first=0
            continue
        fi
        # Trim whitespace
        srv=$(echo "$srv" | tr -d ' ')
        srv=$(echo "$srv" | tr '[:upper:]' '[:lower:]')

        if [[ "$srv" == "$search_server" ]]; then
            case "$col" in
                2) echo $(echo "$p"  | tr -d ' ') ;;
                3) echo $(echo "$u"  | tr -d ' ') ;;
                4) echo $(echo "$pw" | tr -d ' ') ;;
            esac
            return
        fi
    done < "$CREDS_CSV"
}

# ─────────────────────────────────────────────
#  Password cache — avoids prompting twice
#  for the same server
# ─────────────────────────────────────────────
declare -A PWD_CACHE

get_password() {
    local server=$(echo "$1" | tr '[:upper:]' '[:lower:]')

    # Return cached if already known
    if [[ -n "${PWD_CACHE[$server]}" ]]; then
        echo "${PWD_CACHE[$server]}"
        return
    fi

    # Try creds.csv first
    local pwd_csv
    pwd_csv=$(get_cred "$server" "password")

    if [[ -n "$pwd_csv" ]]; then
        PWD_CACHE[$server]="$pwd_csv"
        echo "$pwd_csv"
        return
    fi

    # Not in CSV — prompt user
    local user
    user=$(get_cred "$server" "username")
    local entered
    echo "" >&2
    read -r -s -p "Enter password for server [$server] username [$user]: " entered >&2
    echo "" >&2

    PWD_CACHE[$server]="$entered"
    echo "$entered"
}

# ─────────────────────────────────────────────
#  Build backup filename
#  report.csv -> report_20240315.csv
# ─────────────────────────────────────────────
build_backup_name() {
    local filename="$1"
    local name="${filename%.*}"
    local ext="${filename##*.}"

    if [[ "$filename" == *.* ]]; then
        echo "${name}_${TODAY}.${ext}"
    else
        echo "${filename}_${TODAY}"
    fi
}

# ─────────────────────────────────────────────
#  Check if remote file exists
#  Generates cmds file, runs sftp, checks output
#  Returns 0 if exists, 1 if not
# ─────────────────────────────────────────────
check_remote_file() {
    local server="$1"
    local port="$2"
    local user="$3"
    local password="$4"
    local remote_path="$5"
    local cmds_file="$6"

    # Generate the cmds file
    echo "ls -l $remote_path" >  "$cmds_file"
    echo "bye"                >> "$cmds_file"

    log "INFO" "  Generated: $cmds_file"

    # Run sftp and capture output
    local output
    output=$(sshpass -p "$password" sftp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=10 \
        -P "$port" \
        -b "$cmds_file" \
        "${user}@${server}" 2>/dev/null)

    # If output contains a line starting with "-" it is a file entry
    echo "$output" | grep "^-" > /dev/null 2>&1
    return $?
}

# ─────────────────────────────────────────────
#  Get remote file size in bytes from sftp output
#  Runs sftp ls -l and extracts 5th field (size)
# ─────────────────────────────────────────────
get_remote_size() {
    local server="$1"
    local port="$2"
    local user="$3"
    local password="$4"
    local remote_path="$5"
    local tmp_cmds="$TMP_DIR/cmds_size_$$.txt"

    echo "ls -l $remote_path" >  "$tmp_cmds"
    echo "bye"                >> "$tmp_cmds"

    local output
    output=$(sshpass -p "$password" sftp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=10 \
        -P "$port" \
        -b "$tmp_cmds" \
        "${user}@${server}" 2>/dev/null)

    rm -f "$tmp_cmds"

    # Extract size — 5th field of the ls -l line starting with "-"
    local size_line
    size_line=$(echo "$output" | grep "^-")
    # Use cut to get 5th field (size in bytes)
    echo "$size_line" | tr -s ' ' | cut -d' ' -f5
}

# ─────────────────────────────────────────────
#  Convert bytes to KB
# ─────────────────────────────────────────────
bytes_to_kb() {
    local bytes="$1"
    if [[ -z "$bytes" ]]; then
        echo "N/A"
        return
    fi
    echo "scale=2; $bytes / 1024" | bc
}

# ─────────────────────────────────────────────
#  Rename file on remote server
#  Generates cmds_tgt_rename file and runs sftp
#  Returns 0 on success, 1 on failure
# ─────────────────────────────────────────────
rename_remote_file() {
    local server="$1"
    local port="$2"
    local user="$3"
    local password="$4"
    local src_path="$5"
    local dst_path="$6"
    local cmds_file="$7"

    # Generate the rename cmds file
    echo "-rm $dst_path"          >  "$cmds_file"
    echo "rename $src_path $dst_path" >> "$cmds_file"
    echo "bye"                    >> "$cmds_file"

    log "INFO" "  Generated: $cmds_file"

    sshpass -p "$password" sftp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=30 \
        -P "$port" \
        -b "$cmds_file" \
        "${user}@${server}" >> "$LOG_FILE" 2>&1

    return $?
}

# ─────────────────────────────────────────────
#  Copy file SRC -> local temp -> TGT
#  Returns 0 on success, 1 on failure
# ─────────────────────────────────────────────
copy_file() {
    local src_server="$1"
    local src_port="$2"
    local src_user="$3"
    local src_password="$4"
    local src_path="$5"
    local tgt_server="$6"
    local tgt_port="$7"
    local tgt_user="$8"
    local tgt_password="$9"
    local tgt_path="${10}"
    local tmp_file="$TMP_DIR/xfer_$$.tmp"

    # Pull from SRC to local temp
    log "INFO" "  Pulling from SRC ${src_server}:${src_path}"
    sshpass -p "$src_password" scp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=30 \
        -P "$src_port" \
        "${src_user}@${src_server}:${src_path}" \
        "$tmp_file" >> "$LOG_FILE" 2>&1

    if [[ $? -ne 0 ]]; then
        log "ERROR" "  SCP pull failed from $src_server"
        rm -f "$tmp_file"
        return 1
    fi

    # Push from local temp to TGT
    log "INFO" "  Pushing to TGT ${tgt_server}:${tgt_path}"
    sshpass -p "$tgt_password" scp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=30 \
        -P "$tgt_port" \
        "$tmp_file" \
        "${tgt_user}@${tgt_server}:${tgt_path}" >> "$LOG_FILE" 2>&1

    if [[ $? -ne 0 ]]; then
        log "ERROR" "  SCP push failed to $tgt_server"
        rm -f "$tmp_file"
        return 1
    fi

    rm -f "$tmp_file"
    return 0
}

# ─────────────────────────────────────────────
#  Initialise summary CSV with header
# ─────────────────────────────────────────────
echo "filename,available_in_src,available_in_tgt,backup_filename_in_tgt,file_copied_src_to_tgt,src_file_size_kb,tgt_file_size_kb,backed_up_file_size_kb" \
    > "$SUMMARY_CSV"

# ─────────────────────────────────────────────
#  Counters
# ─────────────────────────────────────────────
TOTAL=0
COPIED=0
BACKED=0
MISSING=0
FAILED=0

# =============================================================
#  MAIN LOOP — read files.csv line by line
# =============================================================
log "INFO" "============================================================"
log "INFO" "SFTP File Transfer Script"
log "INFO" "Run date : $TODAY"
log "INFO" "Folder   : $FOLDER"
log "INFO" "============================================================"

first_line=1
while IFS=',' read -r src_server src_path src_filename tgt_server tgt_path; do

    # Skip header
    if [[ $first_line -eq 1 ]]; then
        first_line=0
        continue
    fi

    # Trim whitespace from all fields
    src_server=$(echo  "$src_server"  | tr -d ' ')
    src_path=$(echo    "$src_path"    | tr -d ' ')
    src_filename=$(echo "$src_filename" | tr -d ' ')
    tgt_server=$(echo  "$tgt_server"  | tr -d ' ')
    tgt_path=$(echo    "$tgt_path"    | tr -d ' ')

    # Remove trailing slash
    src_path="${src_path%/}"
    tgt_path="${tgt_path%/}"

    TOTAL=$((TOTAL + 1))

    src_full="${src_path}/${src_filename}"
    tgt_full="${tgt_path}/${src_filename}"
    bak_name=$(build_backup_name "$src_filename")
    bak_full="${tgt_path}/${bak_name}"

    # cmds file paths for this file
    CMDS_SRC="$TMP_DIR/cmds_src_${src_filename}.txt"
    CMDS_TGT="$TMP_DIR/cmds_tgt_${src_filename}.txt"
    CMDS_TGT_RENAME="$TMP_DIR/cmds_tgt_rename_${src_filename}.txt"

    log "INFO" "────────────────────────────────────────────────────────────"
    log "INFO" "[$TOTAL] $src_filename"
    log "INFO" "  SRC : ${src_server}:${src_full}"
    log "INFO" "  TGT : ${tgt_server}:${tgt_full}"

    # Default summary values
    avail_src="No"
    avail_tgt="No"
    backup_name="N/A"
    copied_flag="No"
    src_size_kb="N/A"
    tgt_size_kb="N/A"
    bak_size_kb="N/A"

    # ── Get credentials ──────────────────────────────────────
    src_port=$(get_cred "$src_server" "port");     src_port=${src_port:-22}
    src_user=$(get_cred "$src_server" "username")
    tgt_port=$(get_cred "$tgt_server" "port");     tgt_port=${tgt_port:-22}
    tgt_user=$(get_cred "$tgt_server" "username")

    if [[ -z "$src_user" ]]; then
        log "ERROR" "  No credentials for SRC: $src_server"
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        FAILED=$((FAILED + 1))
        continue
    fi

    if [[ -z "$tgt_user" ]]; then
        log "ERROR" "  No credentials for TGT: $tgt_server"
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        FAILED=$((FAILED + 1))
        continue
    fi

    # ── Get passwords ─────────────────────────────────────────
    src_password=$(get_password "$src_server")
    tgt_password=$(get_password "$tgt_server")

    # ── 1. Check SRC file ─────────────────────────────────────
    log "INFO" "  Checking SRC file..."
    check_remote_file "$src_server" "$src_port" "$src_user" "$src_password" "$src_full" "$CMDS_SRC"

    if [[ $? -ne 0 ]]; then
        log "WARN" "  Source file NOT found: ${src_server}:${src_full}"
        avail_src="No"
        MISSING=$((MISSING + 1))
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        continue
    fi

    avail_src="Yes"
    src_size=$(get_remote_size "$src_server" "$src_port" "$src_user" "$src_password" "$src_full")
    src_size_kb=$(bytes_to_kb "$src_size")
    log "INFO" "  SRC found. Size: ${src_size_kb} KB"

    # ── 2. Check TGT file ─────────────────────────────────────
    log "INFO" "  Checking TGT file..."
    check_remote_file "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" "$tgt_full" "$CMDS_TGT"

    if [[ $? -eq 0 ]]; then
        avail_tgt="Yes"
        tgt_size=$(get_remote_size "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" "$tgt_full")
        tgt_size_kb=$(bytes_to_kb "$tgt_size")
        log "INFO" "  TGT exists. Size: ${tgt_size_kb} KB. Renaming -> $bak_name"

        # ── 3. Rename TGT file ────────────────────────────────
        rename_remote_file "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" \
            "$tgt_full" "$bak_full" "$CMDS_TGT_RENAME"

        if [[ $? -ne 0 ]]; then
            log "ERROR" "  Rename failed on $tgt_server"
            backup_name="Rename failed"
            echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
            FAILED=$((FAILED + 1))
            continue
        fi

        backup_name="$bak_name"
        bak_size=$(get_remote_size "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" "$bak_full")
        bak_size_kb=$(bytes_to_kb "$bak_size")
        BACKED=$((BACKED + 1))
        log "INFO" "  Rename success. Backup: $bak_name (${bak_size_kb} KB)"

    else
        avail_tgt="No"
        backup_name="File missing in tgt"
        log "INFO" "  No existing file on TGT. Skipping rename."
    fi

    # ── 4. Copy SRC -> TGT ────────────────────────────────────
    log "INFO" "  Copying file..."
    copy_file \
        "$src_server" "$src_port" "$src_user" "$src_password" "$src_full" \
        "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" "$tgt_full"

    if [[ $? -eq 0 ]]; then
        new_tgt_size=$(get_remote_size "$tgt_server" "$tgt_port" "$tgt_user" "$tgt_password" "$tgt_full")
        tgt_size_kb=$(bytes_to_kb "$new_tgt_size")
        copied_flag="Yes"
        COPIED=$((COPIED + 1))
        log "INFO" "  Copy SUCCESS. TGT size: ${tgt_size_kb} KB"
    else
        copied_flag="No"
        FAILED=$((FAILED + 1))
        log "ERROR" "  Copy FAILED for: $src_filename"
    fi

    echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"

done < "$FILES_CSV"

# ─────────────────────────────────────────────
#  Cleanup temp folder
# ─────────────────────────────────────────────
rm -rf "$TMP_DIR"

# ─────────────────────────────────────────────
#  Final results
# ─────────────────────────────────────────────
log "INFO" "============================================================"
log "INFO" "RESULTS : Total=$TOTAL | Copied=$COPIED | Backed=$BACKED | Missing=$MISSING | Failed=$FAILED"
log "INFO" "Summary : $SUMMARY_CSV"
log "INFO" "Log     : $LOG_FILE"
log "INFO" "============================================================"
