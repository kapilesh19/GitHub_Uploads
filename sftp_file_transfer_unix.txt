#!/bin/bash
# =============================================================
#  SFTP File Transfer Script
#  Usage : ./sftp_transfer.sh --folder /path/to/folder
#
#  No sshpass or expect required.
#  Password is prompted naturally by sftp for each connection.
#
#  Folder must contain:
#    files.csv : src_server,src_path,src_filename,tgt_server,tgt_path
#    creds.csv : server_name,port,username
#    (no password column needed in creds.csv)
#
#  Output files generated in same folder:
#    cmds_src_<filename>.txt          check src + get file
#    cmds_tgt_<filename>.txt          check tgt file exists
#    cmds_tgt_rename_<filename>.txt   rename existing tgt file
#    cmds_tgt_put_<filename>.txt      put file to tgt
#    summary_<YYYYMMDD_HHMMSS>.csv
#    sftp_transfer_<YYYYMMDD_HHMMSS>.log
# =============================================================

TIMESTAMP=$(date '+%Y%m%d_%H%M%S')
TODAY=$(date '+%Y%m%d')

# ─────────────────────────────────────────────
#  Parse --folder argument
# ─────────────────────────────────────────────
FOLDER=""
while [[ $# -gt 0 ]]; do
    case "$1" in
        --folder|-f)
            FOLDER="$2"
            shift 2
            ;;
        *)
            echo "Unknown argument: $1"
            echo "Usage: $0 --folder /path/to/folder"
            exit 1
            ;;
    esac
done

if [[ -z "$FOLDER" ]]; then
    echo "ERROR: --folder argument is required"
    echo "Usage: $0 --folder /path/to/folder"
    exit 1
fi

# Convert to absolute path
FOLDER=$(cd "$FOLDER" && pwd)

if [[ ! -d "$FOLDER" ]]; then
    echo "ERROR: Folder not found: $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  File paths
# ─────────────────────────────────────────────
FILES_CSV="$FOLDER/files.csv"
CREDS_CSV="$FOLDER/creds.csv"
SUMMARY_CSV="$FOLDER/summary_${TIMESTAMP}.csv"
LOG_FILE="$FOLDER/sftp_transfer_${TIMESTAMP}.log"
TMP_DIR="$FOLDER/tmp_${TIMESTAMP}"

mkdir -p "$TMP_DIR"

# ─────────────────────────────────────────────
#  Validate input files
# ─────────────────────────────────────────────
if [[ ! -f "$FILES_CSV" ]]; then
    echo "ERROR: files.csv not found in $FOLDER"
    exit 1
fi

if [[ ! -f "$CREDS_CSV" ]]; then
    echo "ERROR: creds.csv not found in $FOLDER"
    exit 1
fi

# ─────────────────────────────────────────────
#  Log function — writes to console and log file
# ─────────────────────────────────────────────
log() {
    local msg="$(date '+%Y-%m-%d %H:%M:%S') [$1] $2"
    echo "$msg"
    echo "$msg" >> "$LOG_FILE"
}

# ─────────────────────────────────────────────
#  Get a field from creds.csv for a server
#  Usage : get_cred <server> <field>
#  Fields: port=2  username=3
# ─────────────────────────────────────────────
get_cred() {
    local search_server=$(echo "$1" | tr '[:upper:]' '[:lower:]')
    local field="$2"
    local col

    case "$field" in
        port)     col=2 ;;
        username) col=3 ;;
    esac

    local first=1
    while IFS=',' read -r srv p u; do
        if [[ $first -eq 1 ]]; then
            first=0
            continue
        fi
        srv=$(echo "$srv" | tr -d ' ' | tr '[:upper:]' '[:lower:]')
        if [[ "$srv" == "$search_server" ]]; then
            case "$col" in
                2) echo $(echo "$p" | tr -d ' ') ;;
                3) echo $(echo "$u" | tr -d ' ') ;;
            esac
            return
        fi
    done < "$CREDS_CSV"
}

# ─────────────────────────────────────────────
#  Build backup filename
#  report.csv -> report_20240315.csv
# ─────────────────────────────────────────────
build_backup_name() {
    local filename="$1"
    if [[ "$filename" == *.* ]]; then
        local name="${filename%.*}"
        local ext="${filename##*.}"
        echo "${name}_${TODAY}.${ext}"
    else
        echo "${filename}_${TODAY}"
    fi
}

# ─────────────────────────────────────────────
#  Convert bytes to KB
# ─────────────────────────────────────────────
bytes_to_kb() {
    local bytes="$1"
    if [[ -z "$bytes" ]]; then
        echo "N/A"
        return
    fi
    echo "scale=2; $bytes / 1024" | bc
}

# ─────────────────────────────────────────────
#  Run sftp with a cmds file
#  sftp naturally prompts for password
#  Usage : run_sftp <user> <server> <port> <cmds_file>
#  Returns 0 on success, 1 on failure
# ─────────────────────────────────────────────
run_sftp() {
    local user="$1"
    local server="$2"
    local port="$3"
    local cmds_file="$4"

    sftp -o StrictHostKeyChecking=no \
         -o ConnectTimeout=30 \
         -P "$port" \
         -b "$cmds_file" \
         "${user}@${server}"

    return $?
}

# ─────────────────────────────────────────────
#  Check if remote file exists
#  Generates cmds file with ls -l
#  Runs sftp and checks output for a file entry
#  Returns 0 if file exists, 1 if not
# ─────────────────────────────────────────────
check_remote_file() {
    local user="$1"
    local server="$2"
    local port="$3"
    local remote_path="$4"
    local cmds_file="$5"

    # Generate cmds file
    echo "ls -l $remote_path" >  "$cmds_file"
    echo "bye"                >> "$cmds_file"

    log "INFO" "  Generated: $cmds_file"

    # Run sftp — password prompted here
    local output
    output=$(sftp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=10 \
        -P "$port" \
        -b "$cmds_file" \
        "${user}@${server}" 2>/dev/null)

    # A line starting with "-" means a regular file was found
    echo "$output" | grep "^-" > /dev/null 2>&1
    return $?
}

# ─────────────────────────────────────────────
#  Get remote file size in bytes
#  Runs sftp ls -l and extracts size with cut
# ─────────────────────────────────────────────
get_remote_size() {
    local user="$1"
    local server="$2"
    local port="$3"
    local remote_path="$4"
    local tmp_cmds="$TMP_DIR/cmds_size_$$.txt"

    echo "ls -l $remote_path" >  "$tmp_cmds"
    echo "bye"                >> "$tmp_cmds"

    local output
    output=$(sftp \
        -o StrictHostKeyChecking=no \
        -o ConnectTimeout=10 \
        -P "$port" \
        -b "$tmp_cmds" \
        "${user}@${server}" 2>/dev/null)

    rm -f "$tmp_cmds"

    # Get the file line, collapse spaces, cut 5th field (size)
    echo "$output" | grep "^-" | tr -s ' ' | cut -d' ' -f5
}

# =============================================================
#  MAIN
# =============================================================
log "INFO" "============================================================"
log "INFO" "SFTP File Transfer Script"
log "INFO" "Run date : $TODAY"
log "INFO" "Folder   : $FOLDER"
log "INFO" "============================================================"

# Write summary CSV header
echo "filename,available_in_src,available_in_tgt,backup_filename_in_tgt,file_copied_src_to_tgt,src_file_size_kb,tgt_file_size_kb,backed_up_file_size_kb" \
    > "$SUMMARY_CSV"

# Counters
TOTAL=0; COPIED=0; BACKED=0; MISSING=0; FAILED=0

# ─────────────────────────────────────────────
#  Read files.csv line by line
# ─────────────────────────────────────────────
first_line=1
while IFS=',' read -r src_server src_path src_filename tgt_server tgt_path; do

    # Skip header
    if [[ $first_line -eq 1 ]]; then
        first_line=0
        continue
    fi

    # Trim whitespace
    src_server=$(echo "$src_server"   | tr -d ' ')
    src_path=$(echo "$src_path"       | tr -d ' ')
    src_filename=$(echo "$src_filename" | tr -d ' ')
    tgt_server=$(echo "$tgt_server"   | tr -d ' ')
    tgt_path=$(echo "$tgt_path"       | tr -d ' ')

    # Remove trailing slash
    src_path="${src_path%/}"
    tgt_path="${tgt_path%/}"

    TOTAL=$((TOTAL + 1))

    src_full="${src_path}/${src_filename}"
    tgt_full="${tgt_path}/${src_filename}"
    bak_name=$(build_backup_name "$src_filename")
    bak_full="${tgt_path}/${bak_name}"

    # cmds file paths for this file
    CMDS_SRC="$TMP_DIR/cmds_src_${src_filename}.txt"
    CMDS_TGT="$TMP_DIR/cmds_tgt_${src_filename}.txt"
    CMDS_TGT_RENAME="$TMP_DIR/cmds_tgt_rename_${src_filename}.txt"
    CMDS_TGT_PUT="$TMP_DIR/cmds_tgt_put_${src_filename}.txt"
    TMP_FILE="$TMP_DIR/xfer_${src_filename}.tmp"

    log "INFO" "────────────────────────────────────────────────────────"
    log "INFO" "[$TOTAL] $src_filename"
    log "INFO" "  SRC : ${src_server}:${src_full}"
    log "INFO" "  TGT : ${tgt_server}:${tgt_full}"

    # Default summary values
    avail_src="No";  avail_tgt="No"
    backup_name="N/A"; copied_flag="No"
    src_size_kb="N/A"; tgt_size_kb="N/A"; bak_size_kb="N/A"

    # ── Get credentials ──────────────────────────────────────
    src_port=$(get_cred "$src_server" "port");   src_port=${src_port:-22}
    src_user=$(get_cred "$src_server" "username")
    tgt_port=$(get_cred "$tgt_server" "port");   tgt_port=${tgt_port:-22}
    tgt_user=$(get_cred "$tgt_server" "username")

    if [[ -z "$src_user" ]]; then
        log "ERROR" "  No credentials for SRC: $src_server"
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        FAILED=$((FAILED + 1)); continue
    fi

    if [[ -z "$tgt_user" ]]; then
        log "ERROR" "  No credentials for TGT: $tgt_server"
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        FAILED=$((FAILED + 1)); continue
    fi

    # ── 1. Check SRC file — sftp prompts password ────────────
    log "INFO" "  Checking SRC file..."
    log "INFO" "  Connecting to SRC: ${src_user}@${src_server}:${src_port}"
    check_remote_file "$src_user" "$src_server" "$src_port" "$src_full" "$CMDS_SRC"

    if [[ $? -ne 0 ]]; then
        log "WARN" "  Source file NOT found: ${src_server}:${src_full}"
        MISSING=$((MISSING + 1))
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        continue
    fi

    avail_src="Yes"
    src_size=$(get_remote_size "$src_user" "$src_server" "$src_port" "$src_full")
    src_size_kb=$(bytes_to_kb "$src_size")
    log "INFO" "  SRC found. Size: ${src_size_kb} KB"

    # ── 2. Check TGT file — sftp prompts password ────────────
    log "INFO" "  Checking TGT file..."
    log "INFO" "  Connecting to TGT: ${tgt_user}@${tgt_server}:${tgt_port}"
    check_remote_file "$tgt_user" "$tgt_server" "$tgt_port" "$tgt_full" "$CMDS_TGT"

    if [[ $? -eq 0 ]]; then
        avail_tgt="Yes"
        tgt_size=$(get_remote_size "$tgt_user" "$tgt_server" "$tgt_port" "$tgt_full")
        tgt_size_kb=$(bytes_to_kb "$tgt_size")
        log "INFO" "  TGT exists. Size: ${tgt_size_kb} KB. Renaming -> $bak_name"

        # ── 3. Rename TGT file ────────────────────────────────
        # Generate rename cmds file
        echo "-rm $bak_full"          >  "$CMDS_TGT_RENAME"
        echo "rename $tgt_full $bak_full" >> "$CMDS_TGT_RENAME"
        echo "bye"                    >> "$CMDS_TGT_RENAME"

        log "INFO" "  Generated: $CMDS_TGT_RENAME"
        log "INFO" "  Connecting to TGT for rename: ${tgt_user}@${tgt_server}:${tgt_port}"

        run_sftp "$tgt_user" "$tgt_server" "$tgt_port" "$CMDS_TGT_RENAME" >> "$LOG_FILE" 2>&1

        if [[ $? -ne 0 ]]; then
            log "ERROR" "  Rename failed on $tgt_server"
            backup_name="Rename failed"
            echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
            FAILED=$((FAILED + 1)); continue
        fi

        backup_name="$bak_name"
        bak_size=$(get_remote_size "$tgt_user" "$tgt_server" "$tgt_port" "$bak_full")
        bak_size_kb=$(bytes_to_kb "$bak_size")
        BACKED=$((BACKED + 1))
        log "INFO" "  Rename success. Backup: $bak_name (${bak_size_kb} KB)"

    else
        avail_tgt="No"
        backup_name="File missing in tgt"
        log "INFO" "  No existing file on TGT. Skipping rename."
    fi

    # ── 4. Get file from SRC to local temp ───────────────────
    log "INFO" "  Getting file from SRC..."

    # Generate cmds_src with get command
    echo "get $src_full $TMP_FILE" >  "$CMDS_SRC"
    echo "bye"                     >> "$CMDS_SRC"

    log "INFO" "  Generated: $CMDS_SRC"
    log "INFO" "  Connecting to SRC for get: ${src_user}@${src_server}:${src_port}"

    run_sftp "$src_user" "$src_server" "$src_port" "$CMDS_SRC" >> "$LOG_FILE" 2>&1

    if [[ $? -ne 0 ]] || [[ ! -f "$TMP_FILE" ]]; then
        log "ERROR" "  GET failed from $src_server"
        rm -f "$TMP_FILE"
        echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"
        FAILED=$((FAILED + 1)); continue
    fi

    log "INFO" "  File pulled to local temp: $TMP_FILE"

    # ── 5. Put file from local temp to TGT ───────────────────
    log "INFO" "  Putting file to TGT..."

    # Generate cmds_tgt_put with put command
    echo "put $TMP_FILE $tgt_full" >  "$CMDS_TGT_PUT"
    echo "bye"                     >> "$CMDS_TGT_PUT"

    log "INFO" "  Generated: $CMDS_TGT_PUT"
    log "INFO" "  Connecting to TGT for put: ${tgt_user}@${tgt_server}:${tgt_port}"

    run_sftp "$tgt_user" "$tgt_server" "$tgt_port" "$CMDS_TGT_PUT" >> "$LOG_FILE" 2>&1

    if [[ $? -ne 0 ]]; then
        log "ERROR" "  PUT failed to $tgt_server"
        rm -f "$TMP_FILE"
        copied_flag="No"
        FAILED=$((FAILED + 1))
    else
        new_tgt_size=$(get_remote_size "$tgt_user" "$tgt_server" "$tgt_port" "$tgt_full")
        tgt_size_kb=$(bytes_to_kb "$new_tgt_size")
        copied_flag="Yes"
        COPIED=$((COPIED + 1))
        log "INFO" "  PUT SUCCESS. TGT size: ${tgt_size_kb} KB"
    fi

    rm -f "$TMP_FILE"

    echo "$src_filename,$avail_src,$avail_tgt,$backup_name,$copied_flag,$src_size_kb,$tgt_size_kb,$bak_size_kb" >> "$SUMMARY_CSV"

done < "$FILES_CSV"

# ─────────────────────────────────────────────
#  Cleanup temp folder
# ─────────────────────────────────────────────
rm -rf "$TMP_DIR"

# ─────────────────────────────────────────────
#  Final results
# ─────────────────────────────────────────────
log "INFO" "============================================================"
log "INFO" "RESULTS : Total=$TOTAL | Copied=$COPIED | Backed=$BACKED | Missing=$MISSING | Failed=$FAILED"
log "INFO" "Summary : $SUMMARY_CSV"
log "INFO" "Log     : $LOG_FILE"
log "INFO" "============================================================"
